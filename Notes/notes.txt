    selector: 'app-child',
    template: `
        <div style="border: 1px solid blue; padding: 10px;">

        <button (click)="sendMessage()">Reply to Parent</button>
        </div>
    `

    export class ChildComponent {
    @Input() parentMessage: string = ''; // Received from Parent
    @Output() replyEvent = new EventEmitter<string>(); // Sent to Parent

    sendMessage() {
        this.replyEvent.emit('Hey Parent, I got your message!');
    }
    }

    import { BehaviorSubject } from 'rxjs';

    @Injectable({ providedIn: 'root' })
    export class DataService {
    // 1. Initialize with a starting value
    private userSource = new BehaviorSubject<string>("Guest");
    
    // 2. Expose as an Observable (for components to subscribe)
    currentUser$ = this.userSource.asObservable();

    // 3. Method to update the value
    updateUser(newName: string) {
        this.userSource.next(newName);
    }
    }

    export class TimerComponent implements OnInit, OnDestroy {
    
    // 1. Create a variable to hold the subscription
    private mySub!: Subscription;

    ngOnInit() {
        // 2. Start a timer that ticks every second
        this.mySub = interval(1000).subscribe(val => {
        console.log('Timer ticking:', val);
        });
    }

    // 3. This fires when the component is removed (e.g., user navigates away)
    ngOnDestroy() {
        console.log('Component being destroyed! Cleaning up...');
        
        // 4. STOP the subscription
        if (this.mySub) {
        this.mySub.unsubscribe();
        }
    }
    }

Pipe Implementation
    import { Pipe, PipeTransform } from '@angular/core';

    @Pipe({
    name: 'reverse',
    standalone: true // Set to true if you are using Standalone Components
    })
    export class ReversePipe implements PipeTransform {

    transform(value: string): string {
        if (!value) return '';
        
        // 1. Split into array: "abc" -> ["a", "b", "c"]
        // 2. Reverse array: ["c", "b", "a"]
        // 3. Join back: "cba"
        return value.split('').reverse().join('');
    }

    }

Directive Implementation
    import { Directive, ElementRef, HostListener, Input } from '@angular/core';

    @Directive({
    selector: '[appHighlight]',
    standalone: true
    })
    export class HighlightDirective {
    // Allow the user to pass a color, default to 'red'
    @Input() appHighlight = 'red';

    constructor(private el: ElementRef) {}

    // Change color on Mouse Enter
    @HostListener('mouseenter') onMouseEnter() {
        this.changeColor(this.appHighlight);
    }

    // Reset color on Mouse Leave
    @HostListener('mouseleave') onMouseLeave() {
        this.changeColor('');
    }

    private changeColor(color: string) {
        this.el.nativeElement.style.color = color;
    }
    }

        const start = (this.currentPage - 1) * this.pageSize;
        const end = start + this.pageSize;
        return this.allData.slice(start, end);

    get totalPages() {
    return Math.ceil(this.allData.length / this.pageSize);
    }
